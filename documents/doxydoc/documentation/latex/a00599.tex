\begin{DoxyParams}[1]{Parametri}
\mbox{\texttt{in}}  & {\em text} & Il testo dal quale estrarre la parola chiave (composto da parole e punti). \\
\hline
\mbox{\texttt{in}}  & {\em word\+Frequencies} & La mappa contenente la frequenza globale di ciascuna parola nella sorgente. \\
\hline
\mbox{\texttt{in}}  & {\em config} & La configurazione del preset contenente le soglie e i limiti correnti. \\
\hline
\end{DoxyParams}
\begin{DoxyReturn}{Restituisce}
La parola estratta idonea alla generazione della domanda. 
\end{DoxyReturn}
\begin{DoxyPrecond}{Precondizione}
Il testo di input, la mappa delle frequenze e la configurazione del preset non devono essere nulli. 
\end{DoxyPrecond}
\begin{DoxyPostcond}{Postcondizione}
La parola restituita rispetta tutti i vincoli di difficoltà o, in caso di esaurimento delle opzioni, rappresenta il miglior compromesso (fallback). 
\end{DoxyPostcond}
